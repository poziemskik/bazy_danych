ZAD 1
CREATE TABLE CLIENT AS SELECT * FROM company_2025.CLIENT;
CREATE TABLE `ORDER` AS SELECT * FROM company_2025.`ORDER`;
CREATE TABLE EMPLOYEE AS SELECT * FROM company_2025.EMPLOYEE;

SELECT * FROM `ORDER`;

SELECT * FROM `ORDER`
WHERE ORDER_STATUS = 5;

SELECT ORDER_ID, ORDER_STATUS
FROM `ORDER`
WHERE CLIENT_ID IN (1, 3, 5);



ZAD 2
SELECT *
FROM EMPLOYEE
WHERE FIRST_NAME = 'Agnieszka'
AND pensja = 3900;

SELECT *
FROM EMPLOYEE
WHERE pensja BETWEEN 5000 AND 7000;

SELECT *
FROM CLIENT
WHERE LOWER(SHORT_NAME) LIKE '%eco%'
AND IS_COMPANY = 1;



ZAD 3
SELECT *
FROM `ORDER`
WHERE MONTH(ORDER_DATE) IN (6, 7);

SELECT *
FROM CLIENT
WHERE TAX_IDENTIFIER IS NOT NULL 
ORDER BY SHORT_NAME ASC;

SELECT *
FROM EMPLOYEE
ORDER BY BIRTH_DATE ASC 
LIMIT 5;



ZAD 4
SELECT DISTINCT FIRST_NAME
FROM EMPLOYEE;

SELECT CONCAT(FIRST_NAME, ' - ', LAST_NAME) AS imie_nazwisko
FROM EMPLOYEE
WHERE LAST_NAME LIKE 'Ko%';

SELECT LAST_NAME, pensja
FROM EMPLOYEE
WHERE DATE_OF_EMPLOYMENT >= '2010-01-01' 
AND DATE_OF_EMPLOYMENT <  '2015-01-01';

SELECT
EMPLOYEE_NAME,
FIRST_NAME,
LAST_NAME,
pensja AS pensja_brutto, 
ROUND(pensja * 0.75, 2) AS pensja_netto,
ROUND(pensja * 0.25, 2) AS podatek
FROM EMPLOYEE;
